set(CMAKE_BUILD_TYPE "Release")

    "*.h"
	"*.c"
    "*.cpp"
)

	"${MAX_SDK_INCLUDES}"
	"${MAX_SDK_MSP_INCLUDES}"
	"${MAX_SDK_JIT_INCLUDES}"
)

add_library( 
	${PROJECT_NAME}

	${PROJECT_SRC}
)

	${PROJECT_NAME}
	PUBLIC
	${CMAKE_CURRENT_SOURCE_DIR}
    ${THIRDPARTY_DEPS}/pocketpy
)

target_link_libraries(
    ${PROJECT_NAME}
    PUBLIC
    pocketpy
)
